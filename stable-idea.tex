\begin{frame}{What is a stable model?}
  \begin{itemize}
  \item \structure{Lessons learned} from positive programs
    \begin{itemize}
    \item Every atom true in \Cn{P} is justified by a proof
    \item This yields the smallest model of $P$, eliminating all non-minimal ones
    \end{itemize}
  \item \structure{Procederal attempt}
      \item [] \textbf{let} $\T{P}{X}=\{\head{r}\mid r\in P, \pbody{r}\subseteq X\}$ \textbf{in}
    \begin{itemize}\normalsize
    \item []$X := \emptyset$
      \smallskip
    \item []\textbf{while} {$\T{P}{X}\neq X$}
      \begin{itemize}\normalsize
      \item[] $X := {\,\T{P}{X}}$
      \end{itemize}
    \item[] \textbf{return} $X$
    \end{itemize}
    \item \structure{Example} \
      $P=\{a\leftarrow{},b\leftarrow \neg c,c\leftarrow b\}$

  \item \structure{Logical attempt}
  \item \structure{Example} \
          $P=\{a\leftarrow\neg b\}$

  \end{itemize}
\end{frame}

    \alert<5>{\{p,q\}}

\uncover<7->{%
Informally, a set $X$ of atoms is a \alert{stable model} of a logic program $P$
\begin{itemize}
\item if $X$ is a (classical) model of $P$
  and
\item if all atoms in $X$ are \alert{justified} by some rule in $P$
\end{itemize}
% \pause
% {\small (rooted in intuitionistic logics HT~(Heyting,~1930) and G3~(G\"odel,~1932))}
