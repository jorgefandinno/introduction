\begin{frame}<handout:14,15>{Example one}
% ------------------------------
\newcommand{\PIip}{%
  \begin{array}[t]{lll}
    p &\leftarrow&p
    \\
      &&
  \end{array}}
\newcommand{\PIop}{%
  \begin{array}[t]{lll}
    p &\leftarrow&p
    \\
    q &\leftarrow&
  \end{array}}
% ------------------------------
\begin{itemize}
\item<1->
  \(
  P
  =
  \left\{
    p \leftarrow p, \
    q \leftarrow \neg p
  \right\}
  \)
  \medskip
\item<2->[]
  \[
    \begin{array}{c|cl|ccr}
      X                          &\ &\uncover< 3-13>{\reduct{P}{X}}&\ &\uncover< 3-13>{\Cn{\reduct{P}{X}}}&                                  \\\hline
      \{\phantom{p         ,q}\} &  &\uncover< 4-13>{\PIop}        &  &\uncover< 5-13>{\{q\}}             &\uncover< 6->              {\KO}  \\\hline
      \{         p\phantom{,q}\} &  &\uncover< 7-13>{\PIip}        &  &\uncover< 8-13>{\emptyset}         &\uncover< 9->{\alt<all:15>{\OK}{\KO}} \\\hline
      \{\phantom{p,}        q \} &  &\uncover<10-13>{\PIop}        &  &\uncover<10-13>{\{q\}}             &\uncover<11->              {\OK}  \\\hline
      \{         p         ,q \} &  &\uncover<12-13>{\PIip}        &  &\uncover<12-13>{\emptyset}         &\uncover<13->{\alt<all:15>{\OK}{\KO}}
    \end{array}
  \]
  \medskip
\item<6->[] \OK\quad\only<all:-14>{ stable} model\only<all:15->{\phantom{ stable}} \qquad \KO\quad no\only<all:-14>{ stable} model
\end{itemize}
\end{frame}
